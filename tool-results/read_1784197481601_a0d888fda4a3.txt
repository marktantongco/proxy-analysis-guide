    45→  const isDark = theme === "dark";
    46→
    47→  return (
    48→    <Button
    49→      variant="ghost"
    50→      size="sm"
    51→      onClick={() => setTheme(isDark ? "light" : "dark")}

    54→      <AnimatePresence mode="wait">
    55→        {isDark ? (
    56→          <motion.div key="sun" initial={{ rotate: -90, opacity: 0 }} animate={{ rotate: 0, opacity: 1 }} exit={{ rotate: 90, opacity: 0 }} transition={{ duration: 0.2 }}>
    57→            <Sun className="h-4 w-4 text-[#ae964d]" />

   137→export function ComponentSearchSection() {
   138→  const ref = useRef(null);
   139→  const isInView = useInView(ref, { once: true, margin: "-50px" });
   140→  const [query, setQuery] = useState("");
   141→  const [results, setResults] = useState<SearchResult[]>([]);
   142→  const [loading, setLoading] = useState(false);
   143→  const [totalPages, setTotalPages] = useState(0);
   144→  const [currentPage, setCurrentPage] = useState(1);
   145→  const [requestsLeft, setRequestsLeft] = useState<number | null>(null);
   146→
   147→  const search = useCallback(async (searchQuery: string, page = 1) => {
   148→    if (!searchQuery.trim()) { setResults([]); return; }
   149→    setLoading(true);
   150→    try {
   151→      const res = await fetch("/api/components?XTransformPort=3030", {
   152→        method: "POST",
   153→        headers: { "Content-Type": "application/json" },
   154→        body: JSON.stringify({ search: searchQuery, page, per_page: 12 }),
   155→      });
   156→      // Fallback: try direct 21st.dev API route (no port transform needed for Next.js API)
   157→      const data = await res.json();
   158→      setResults(data.results || []);
   159→      if (data.metadata) {
   160→        setTotalPages(data.metadata.pagination?.total_pages || 0);
   161→        setRequestsLeft(data.metadata.requests_remaining);
   162→      }
   163→    } catch {
   164→      // Try without port transform
   165→      try {
   166→        const res2 = await fetch("/api/components", {
   167→          method: "POST",
   168→          headers: { "Content-Type": "application/json" },
   169→          body: JSON.stringify({ search: searchQuery, page, per_page: 12 }),
   170→        });
   171→        const data = await res2.json();
   172→        setResults(data.results || []);
   173→        if (data.metadata) {
   174→          setTotalPages(data.metadata.pagination?.total_pages || 0);
   175→          setRequestsLeft(data.metadata.requests_remaining);
   176→        }
   177→      } catch {
   178→        setResults([]);
   179→      }
   180→    }
   181→    setLoading(false);
   182→  }, []);
   183→
   184→  const handleSearch = useCallback(() => {
   185→    setCurrentPage(1);
   186→    search(query, 1);
   187→  }, [query, search]);
   188→
   189→  return (
   190→    <section ref={ref} id="components" className="py-12 sm:py-20 bg-white dark:bg-[#1a1a1e]">
   191→      <div className="max-w-5xl mx-auto px-4 sm:px-6">
   192→        <motion.div variants={fadeUp} initial="hidden" animate={isInView ? "visible" : "hidden"}>
   193→          <div className="flex items-center justify-between flex-wrap gap-2 mb-2">
   194→            <h2 className="text-2xl sm:text-3xl font-bold text-[#6b634d] dark:text-[#ae964d]">Component Explorer</h2>
   195→            <Badge variant="outline" className="text-xs border-[#537ba4]/30 text-[#537ba4]">
   196→              <Package className="h-3 w-3 mr-1" /> Powered by 21st.dev
   197→            </Badge>
   198→          </div>
   199→          <p className="text-[#86837c] dark:text-[#a0a0a0] mb-6 text-sm sm:text-base">
   200→            Search 1400+ React UI components from the 21st.dev registry. Find, preview, and install with one command.
   201→          </p>
   202→        </motion.div>
   203→
   204→        <motion.div variants={fadeUp} initial="hidden" animate={isInView ? "visible" : "hidden"}>
   205→          <div className="flex gap-2 mb-4">
   206→            <div className="flex-1 relative">
   207→              <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-[#86837c] dark:text-[#a0a0a0]" />
   208→              <Input
   209→                placeholder="Search components (button, card, dialog...)"
   210→                value={query}
   211→                onChange={(e) => setQuery(e.target.value)}
   212→                onKeyDown={(e) => e.key === "Enter" && handleSearch()}
   213→                className="pl-10 bg-[#f6f6f6] dark:bg-[#2a2a2e] border-[#d6d1c2] dark:border-[#3a3a3e] text-[#151513] dark:text-[#e8e6e1]"
   214→              />
   215→            </div>
   216→            <Button onClick={handleSearch} disabled={loading} className="bg-[#6b634d] hover:bg-[#5a5540] text-white">
   217→              {loading ? <RefreshCw className="h-4 w-4 animate-spin" /> : <Search className="h-4 w-4" />}

   226→            </div>

   228→        </motion.div>
   229→
   230→        <motion.div variants={staggerContainer} initial="hidden" animate={isInView ? "visible" : "hidden"} className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-3">
   231→          {results.map((result, i) => (
   232→            <motion.div key={`${result.component_data.component_slug}-${result.demo_id || i}`} variants={staggerItem}>
   233→              <ComponentCard result={result} />

   237→
   238→        {results.length === 0 && query && !loading && (
   239→          <div className="text-center py-8 text-[#86837c] dark:text-[#a0a0a0]">
   240→            <Package className="h-12 w-12 mx-auto mb-2 opacity-30" />

   243→          </div>
   244→        )}
   245→
   246→        {totalPages > 1 && (
   247→          <div className="flex items-center justify-center gap-2 mt-6">
   248→            <Button
   249→              variant="outline"
   250→              size="sm"
   251→              disabled={currentPage <= 1}
   252→              onClick={() => { const p = currentPage - 1; setCurrentPage(p); search(query, p); }}
   253→              className="border-[#d6d1c2] dark:border-[#3a3a3e]"
