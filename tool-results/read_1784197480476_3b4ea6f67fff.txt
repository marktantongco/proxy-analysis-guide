    68→function scoreColor(score: number) {
    69→  if (score >= 9) return "text-[#3f7450]";
    70→  if (score >= 7.5) return "text-[#537ba4]";
    71→  if (score >= 6.5) return "text-[#ae8d4a]";
    72→  return "text-[#8b4e49]";

    75→function scoreBg(score: number) {
    76→  if (score >= 9) return "bg-[#3f7450]/10 border-[#3f7450]/30";

    84→function useCopy() {
    85→  const [copied, setCopied] = useState(false);
    86→  const copy = useCallback((text: string) => {
    87→    navigator.clipboard.writeText(text);

    89→    setTimeout(() => setCopied(false), 2000);
    90→  }, []);
    91→  return { copied, copy };
